  /**
   * Deterministic Fisher-Yates Shuffle
   * @method _shuffleArr
   * @memberof Cripta
   * @instance
   * @private
   */
  _shuffleArr: function(array, mask, reverse = false) {
    const result = [...array];
    const n = result.length;

    // Mapping bytes to indices
    if (!reverse) {
      for (let i = n - 1; i > 0; i--) {
        const j = mask[mask.length - 1 - i] % (i + 1);
        [result[i], result[j]] = [result[j], result[i]];
      }
    } else {
      for (let i = 1; i < n; i++) {
        const j = mask[mask.length - 1 - i] % (i + 1);
        [result[i], result[j]] = [result[j], result[i]];
      }
    }
    return result;
  },

  _generateString: function(buff, charset, type, length) {

    const charLen = charset.length;
    // Calculate the max byte to discard to prevent bias
    const maxValidByte = Math.floor(256 / charLen) * charLen - 1;

    let pool = this._getEntropy(length * 2, buff, type);
    let offset = 0;

    while (res.length < length) {
      if (offset >= pool.length) {
        pool = this._getEntropy(length, buff, pool);
        offset = 0;
      }
      const b = pool[offset++];
      if (b <= maxValidByte) {
        res += charset[b % charLen];
      }
    }
    pool.fill(0);
    return res;
  },

  /**
   * Executes an Argon2id memory-hard key derivation.
   * Argon2id is used to provide maximum resistance against GPU/ASIC brute-force 
   * attacks by enforcing a configurable memory and CPU cost.
   * @method _argon2
   * @memberof Cripta
   * @instance
   * @private
   */
  _argon2: function(message, nonce, callback = false, _memory = 0) {
    if (!_memory || _memory < 4 * 1024) throw new Error('Memory can not be empty or less than 4KiB.');

    // for lite mode
    let memory = 32 * 1024;           // 32 MiB
    let tagLength = 4 * 1024;         // 4KB output length

    if (!this.constants._liteMode) {
      tagLength = Math.floor(_memory * 0.5);   // bare minimum
      if (tagLength % 2 !== 0) tagLength++;
      memory = _memory + 0;
    }

    return new Promise((resolve, reject) => {
      const { raw, passes, parallelism } = this.constants.argon2;
      crypto.argon2('argon2id', {
        message: Buffer.isBuffer(message) ? message : Buffer.from(message, 'utf8'),
        nonce: Buffer.isBuffer(nonce) ? nonce : Buffer.from(nonce, 'utf8'),
        raw, memory, passes, parallelism, tagLength
      }, (err, derivedKey) => {
        callback && callback();
        err ? reject(err) : resolve(derivedKey);
      });
    });
  },

  /**
   * Formats raw entropy into Mnemonic, and TV-Code CID representations.
   * @method _formatCID
   * @memberof Cripta
   * @instance
   * @private
   */
  _formatCID: function(entropyBuf, isClear, disableChecksum = false, obfuscateFormat = false) {
    let checksum, fullPayload;
    if (!obfuscateFormat && !this.constants._bitStandardCID.includes(entropyBuf.length * 8))
      throw new Error(`INVALID CID: Invalid standard CID size (${entropyBuf.length * 8}).`);
    if (disableChecksum) {
      fullPayload = Buffer.from(entropyBuf);
    } else {
      checksum = this._getChecksum(entropyBuf);
      fullPayload = Buffer.concat([entropyBuf, checksum]);
    }
    let { words } = this._constants.cache;
    let tvCodeFormat = this._encodeTVcode(fullPayload);
    let mArray = [];
    for (let i = 0; i < fullPayload.length; i += 2) {
      mArray.push(words[fullPayload.readUInt16BE(i)]);
    }
    if (isClear) {
      checksum && checksum.fill(0);
      fullPayload.fill(0);
      entropyBuf.fill(0);
      entropyBuf = null;
    }
    return {
      buffer: fullPayload,
      tvCode: tvCodeFormat,
      mnemonic: mArray.join(' ')
    };
  },

  /**
   * Validates and parses a Public Nonce ID (NID) string.
   * Enforces normalization (NFC) and strict regex filtering.
   * @method _formatNID
   * @memberof Cripta
   * @instance
   * @private
   */
  _formatNID: function(nid) {
    if (!nid || typeof nid !== 'string') throw new Error('NID must be a string.');

    const parts = nid.trim().split(':');
    if (parts.length !== 4) throw new Error('Invalid NID format: Must have 4 segments.');  

    // Normalize the label to NFC (Canonical Composition) immediately.
    // This resolves homoglyphs that are technically "decomposed" vs "composed".
    const label = parts[0].normalize('NFC'); 
    const [_, rawNonce, type, lengthStr] = parts;

    if (!label || label.length > this._constants.maxLabelLength) throw new Error(`Label must be 1-${this._constants.maxLabelLength} characters.`);

    const labelRegex = /^[\p{L}\p{N}_\-@.[\]{}()#+~]+$/u;
    const hasAlpha = /\p{L}/u.test(label);

    if (!hasAlpha || !labelRegex.test(label))
      throw new Error('Invalid Label: Must contain letters, no spaces/special commands.');
    if (!rawNonce.includes('-'))
      throw new Error('Invalid NID: Nonce must contain dashes (e.g., ABCD-1234).');
    const cleanNonce = rawNonce.replace(/-/g, '');
    if (!/^[A-F0-9]+$/i.test(cleanNonce) || cleanNonce.length < 8 || cleanNonce.length > 20 || cleanNonce.length % 2 !== 0)
      throw new Error('Invalid NID: Nonce length must be between 8 and 20 hex characters.');
    if (!this._constants.pwTypesFirstChar.includes(type.charAt(0)))
      throw new Error('Invalid NID: Unknown password type.');

    const validModes = ['AES', 'CHACHA'];
    let length = 0;
    // If not authenticated encryption mode, it must be a numerical length.
    if (!validModes.includes(lengthStr.toUpperCase())) {
      if (!/^\d+$/.test(lengthStr))
        throw new Error('Invalid NID: Length must be a valid numerical value and only contain digits.');
      if (lengthStr.startsWith('0') && lengthStr.length > 1)
        throw new Error(`Invalid NID: Length detected leading zeroes. (${lengthStr})`);
      if (lengthStr.length > 15)
        throw new Error('Invalid NID: Length exceeds maximum allowable magnitude.');

      length = parseInt(lengthStr, 10);
      if (!Number.isSafeInteger(length) || isNaN(length))
        throw new Error('Invalid NID: Length must be a valid safe integer.');
      if (length < 1 || length > this.constants._maxOutputLength)
        throw new Error(`Invalid NID: Length out of bounds (1-${this.constants._maxOutputLength}).`);
    }
    return [cleanNonce, type.trim(), lengthStr.trim(), length];
  },

  /**
   * Normalizes and pads a user-provided PIN to 16 digits.
   * @method _formatPIN
   * @memberof Cripta
   * @instance
   * @private

  _formatPIN: function (str) {
    if (!/^\d+$/.test(str))
      throw new Error("Invalid input: PIN must only contain digits.");
    if (str.length > 16)
      throw new Error("Invalid input: PIN exceeds 16-digit limit.");
    if (str.startsWith('0') && str.length > 1)
      throw new Error(`Invalid input: PIN detected leading zeroes. (${str})`);
    return str.padStart(16, '0');

   * Validates Cryptographic Salt.
   * Strict ASCII-only enforcement and lowercased to prevent cross-platform encoding drift and human error.
   * @method _formatSalt
   * @memberof Cripta
   * @instance
   * @private

  _formatSalt: function (salt) {
    if (!salt || typeof salt !== 'string') throw new Error('Salt must be a string.');
    const trimmed = salt.trim();
    if (trimmed.toLowerCase() !== salt) throw new Error('Salt must always be lowercase, detected some UPPERCASE which can lead to long-term memory risks.');
    if (trimmed.length < 12) throw new Error('Salt must not be empty or less than 12 long.');

    // Strict Whitespace Check (No spaces allowed anywhere)
    if (/\s/.test(trimmed))
      throw new Error('Salt must not contain spaces, tabs, or line breaks.');
    if (/[^\x20-\x7E]/.test(trimmed))
      throw new Error('Salt contains invalid characters. Use only standard letters, numbers, and symbols (!@#$).');
    if (trimmed.length > this._constants.maxSaltLength)
      throw new Error(`Salt length exceeds maximum limit of ${this._constants.maxSaltLength}`);
    return trimmed;
  },

  /**

   * @method _verifyWords

  _verifyWords: function(inputWords, fullBuffer = null) {
    if (!inputWords) return;

    const { words, wordMap } = this._constants.cache;
    let inputW = (typeof inputWords === 'string') ? inputWords.split(/\s+/) : inputWords;

      if (typeof index === 'undefined') throw new Error(`Word "${word}" not found!`);
      fullBuffer && fullBuffer.writeUInt16BE(index, i * 2);
    });
  },

  _defineSecret: async function(newfullBuffer, _type, length) {
    const type = _type.trim();
    const checkInvalidType = (optn = 1) => { if (type.length > optn) throw new Error(`Invalid type: Extensions for '${type.charAt(0)}' are not supported. Received: ${type}`); };
    const { words, wordMap } = this._constants.cache;
    let generatedOutput;

    try {
      switch (type.charAt(0)) {
        case 'R': { // Raw Bytes
          checkInvalidType();
          generatedOutput = this._getEntropy(length, newfullBuffer, type); break;
        }
        case 'H': { // Hex String
          checkInvalidType();
          generatedOutput = this._getEntropy(length, newfullBuffer, type).toString('hex'); break;
        }
        case 'B': { // Base64 or Base58 String
          if (type === 'B5') {
            generatedOutput = bs58.encode(new Uint8Array(this._getEntropy(length, newfullBuffer, type).buffer));
          } else {
            checkInvalidType();
            generatedOutput = this._getEntropy(length, newfullBuffer, type).toString('base64');
          }
          break;
        }
        case 'L': { // Letters-only (A-Z)
          checkInvalidType();
          generatedOutput = this._generateString(newfullBuffer, "ABCDEFGHIJKLMNOPQRSTUVWXYZ", type, length);
          break;
        }
        case 'N': { // Numeric-only (0-9)
          checkInvalidType();
          generatedOutput = this._generateString(newfullBuffer, "0123456789", type, length);
          break;
        }
        case 'P': { // Password
          checkInvalidType();
          generatedOutput = this._generateString(newfullBuffer, this._constants.charsP, type, length);
          break;
        }
        case 'M': { // Mnemonic Password (Supports chars like M_, M-, M+, M=)
          checkInvalidType(2);

          let pMnemonic = [];
          let pool = this._getEntropy(length * 2, newfullBuffer, type);
          for (let j = 0; j < length; j++) {
            const index = pool.readUInt16BE(j * 2);
            pMnemonic.push(words[index % words.length]);
          }

          let separator = ' ';
          if (type.length === 2) {
            const charCode = type.charCodeAt(1);
            // Limit to Printable ASCII: 32 (Space) to 126 (Tilde ~)
            if (charCode < 32 || charCode > 126)
              throw new Error(`Invalid Mnemonic separator: '${type.charAt(1)}'. Please use a printable ASCII character.`);
            separator = type.charAt(1);
          }

          generatedOutput = pMnemonic.join(separator);
          pool.fill(0);
          break;
        }

        case 'W': { // Wallet Case
          let entropy;
          // Bitcoin WIF (Compressed / Modern Default)
          if (type === 'WF') {
            if (length !== 32) throw new Error(`Invalid length for Bitcoin WIF. Expected 32 bytes, but got ${length} bytes.`);
            entropy = this._getEntropy(length, newfullBuffer, "BTC");
            generatedOutput = wif.encode({
              version: 128, // (0x80) Bitcoin Mainnet Prefix
              privateKey: entropy, 
              compressed: true
            });

          // Standard BIP39 Mnemonic
          } else {
            checkInvalidType();
            if (![16, 24, 32].includes(length)) throw new Error(`Invalid length for BIP39 seed. Please use either 16, 24, or 32 bytes. Got ${length} bytes.\n Map word count to entropy bytes:\n\n  12 words = 16 bytes\n  18 words = 24 bytes\n  24 words = 32 bytes`);
            entropy = this._getEntropy(length, newfullBuffer, "BIP39");
            generatedOutput = bip39.entropyToMnemonic(entropy);
          }

          entropy.fill(0);
          break;
        }

        case 'E': { // EVM Private Key for Ethereum, BSC, MATIC, C-Chain, Arbitrum/Optimism, etc.
          if (length !== 32) throw new Error(`Invalid length for EVM private key. Expected 32 bytes, but got ${length} bytes.`);
          checkInvalidType();

          const totalAttempts = 8;
          const pool = this._getEntropy(length * totalAttempts, newfullBuffer, "EVM");
          let failureCount = 0;
          let isValid = false;

          try {
            while (!isValid) {
              if (failureCount >= totalAttempts)
                throw new Error('Wall of bad entropy exceeded. Please try incrementing your PIN number or generate another nonce for your NID.');

              const start = failureCount * length;
              const buffer = pool.subarray(start, start + length);
              const privateKeyHex = "0x" + buffer.toString('hex');

              try {
                // ethers.Wallet acts as "Oracle" validator
                let wallet = new ethers.Wallet(privateKeyHex);

                if (wallet.address && wallet.address.startsWith('0x')) {
                  generatedOutput = privateKeyHex;
                  isValid = true;
                  wallet = null;
                } else {
                  failureCount++;
                }
              } catch (e) {
                // Usually fails if entropy is outside secp256k1 curve order
                failureCount++;
              }
            }
          }
          catch (err) { throw err; }
          finally { pool.fill(0); }
          break;
        }

        case 'S': {
          // --- Session Messenger (EXPERIMENTAL) ---
          // https://github.com/session-foundation/session-desktop/blob/dev/ts/session/crypto/mnemonic.ts
          if (type === 'SE') {
            if (length !== 16) throw new Error(`Invalid length for Session recovery password. Expected 16 bytes, but got ${length} bytes.`); 
            checkInvalidType(2);

            const wordset = sessionWordlList;
            const n = wordset.length;
            const prefixLen = 3;
            const entropy = this._getEntropy(length, newfullBuffer, "SESSION-MESSENGER");

            // Swap Endianness (4-byte blocks)
            let swappedHex = "";
            const hexEntropy = entropy.toString('hex');
            for (let j = 0; j < hexEntropy.length; j += 8) {
                const chunk = hexEntropy.slice(j, j + 8);
                swappedHex += chunk.slice(6, 8) + chunk.slice(4, 6) + chunk.slice(2, 4) + chunk.slice(0, 2);
            }

            // Encode words using the Base-1626 math
            let out = [];
            for (let i = 0; i < swappedHex.length; i += 8) {
                const x = parseInt(swappedHex.substr(i, 8), 16);
                const w1 = x % n;
                const w2 = (Math.floor(x / n) + w1) % n;
                const w3 = (Math.floor(Math.floor(x / n) / n) + w2) % n;
                out.push(wordset[w1], wordset[w2], wordset[w3]);
            }

            // Calculate Checksum Index (CRC32 of 3-letter prefixes)
            let trimmedWords = "";
            for (let i = 0; i < out.length; i++) {
                trimmedWords += out[i].slice(0, prefixLen);
            }
            const checksum = crc32.unsigned(Buffer.from(trimmedWords));
            const checksumIndex = checksum % out.length;
            out.push(out[checksumIndex]);

            generatedOutput = out.join(' ');
            entropy.fill(0);
            break;
          }

          // Solana Wallet
          if (length !== 32) throw new Error(`Invalid length for Solana seed. Expected 32 bytes, but got ${length} bytes.`); 
          checkInvalidType();

          const totalAttempts = 8;
          const pool = this._getEntropy(length * totalAttempts, newfullBuffer, "SOL");
          let failureCount = 0;
          let isValid = false;

          try {
            while (!isValid) {
              if (failureCount >= totalAttempts)
                throw new Error('Wall of bad entropy exceeded. Please try incrementing your PIN number or generate another nonce for your NID.');

              const start = failureCount * length;
              const buffer = pool.subarray(start, start + length);

              if (buffer.every(b => b === 0) || buffer.every(b => b === 255)) {
                failureCount++;
              } else {
                const keypair = Keypair.fromSeed(buffer);
                generatedOutput = bs58.encode(keypair.secretKey);
                if (keypair.secretKey instanceof Uint8Array) keypair.secretKey.fill(0);
                isValid = true;
              }
            }
          } catch (err) { throw err; }
          finally { pool.fill(0); }
          break;
        }

        case 'T': { // TRON Private Key
          if (length !== 32) throw new Error(`Invalid length for TRON private key. Expected 32 bytes, but got ${length} bytes.`);
          checkInvalidType();

          const totalAttempts = 8;
          const buffer = this._getEntropy(length * totalAttempts, newfullBuffer, "TRX");
          let failureCount = 0;
          let isValid = false;

          try {
            while (!isValid) {
              if (failureCount >= totalAttempts)
                throw new Error('Wall of bad entropy exceeded. Please try incrementing your PIN number or generate another nonce for your NID.');

              // Grab the next 32-byte chunk
              const start = failureCount * length;
              const checkBuf = buffer.subarray(start, start + length);

              // Check for Zero Key (Invalid)
              if (checkBuf.some(b => b !== 0)) { 
                generatedOutput = checkBuf.toString('hex').toUpperCase();
                isValid = true;
              } else {
                failureCount++;
              }
            }
          }
          catch (err) { throw err; }
          finally { buffer.fill(0); }
          break;
        }

        case 'A': { // Algorand
          if (length !== 32) throw new Error(`Invalid length for Algorand seed. Expected 32 bytes, but got ${length} bytes.`);
          checkInvalidType();

          const totalAttempts = 8;
          const pool = this._getEntropy(length * totalAttempts, newfullBuffer, "ALGO");
          let failureCount = 0;
          let isValid = false;

          try {
            while (!isValid) {
              if (failureCount >= totalAttempts)
                throw new Error('Wall of bad entropy exceeded. Please try incrementing your PIN number or generate another nonce for your NID.');

              const start = failureCount * length;
              const buffer = pool.subarray(start, start + length);
              const seed = new Uint8Array(buffer);

              try {
                // Oracle: Generate the mnemonic and verify the address
                const mnemonic = algosdk.mnemonicFromSeed(seed);
                const account = algosdk.mnemonicToSecretKey(mnemonic);

                if (algosdk.isValidAddress(account.addr.toString())) {
                  generatedOutput = mnemonic;
                  isValid = true;
                } else {
                  failureCount++;
                }
              } catch (e) {
                // Fails if the seed is mathematically invalid for the SDK
                failureCount++;
              }
            }
          } 
          catch (err) { throw err; }
          finally { pool.fill(0); }
          break;
        }

        case 'X': { // Monero (Legacy) Wallet
          if (length !== 32) throw new Error(`Invalid length for Monero legacy seed. Expected 32 bytes, but got ${length} bytes.`);
          checkInvalidType();

          const tempPassword = this._rnd(64).toString('base64');
          const n = xmrWordlList.length;
          const totalAttempts = 3;

          const pool = this._getEntropy(length * totalAttempts, newfullBuffer, "XMR");
          let failureCount = 0;
          let isValid = false;

          try {
            while (!isValid) {
              if (failureCount >= totalAttempts)
                throw new Error('Wall of bad entropy exceeded. Please try incrementing your PIN number or generate another nonce for your NID.');

              const mnemonicArray = [];
              let wallet;

              try {
                // Use zero-copy subarray for the current 32-byte window
                const start = failureCount * length;
                const buffer = pool.subarray(start, start + length);

                // 24-word math (4 bytes -> 3 words)
                for (let i = 0; i < buffer.length; i += 4) {
                  const val = buffer.readUInt32LE(i);
                  const w1 = val % n;
                  const w2 = (Math.floor(val / n) + w1) % n;
                  const w3 = (Math.floor(val / (n * n)) + w2) % n;
                  mnemonicArray.push(xmrWordlList[w1], xmrWordlList[w2], xmrWordlList[w3]);
                }

                // CRC32 Checksum for 25th word
                let checksumString = "";
                for (const word of mnemonicArray) checksumString += word.substring(0, 3);
                const checkIndex = crc32.unsigned(checksumString) % mnemonicArray.length;
                mnemonicArray.push(mnemonicArray[checkIndex]);
                const candidateSeed = mnemonicArray.join(' ');

                wallet = await moneroTs.createWalletKeys({
                  networkType: moneroTs.MoneroNetworkType.MAINNET,
                  seed: candidateSeed,
                  password: tempPassword
                });

                const address = await wallet.getPrimaryAddress();
                const confirmedSeed = await wallet.getSeed();
                await wallet.close(false);

                if (address && address.startsWith('4') && confirmedSeed === candidateSeed) {
                  generatedOutput = candidateSeed;
                  isValid = true;
                } else {
                  failureCount++;
                }
              } catch (e) {
                if (wallet) await wallet.close(false);
                failureCount++;
              }
            }
          }
          catch (err) { throw err; }
          finally {
            pool.fill(0);
            try { await moneroTs.shutdown(); } catch (_) {}
          }
          break;
        }

        case 'D': { // Polkadot (DOT)
          if (length !== 32) throw new Error(`Invalid length for Polkadot seed. Expected 32 bytes, but got ${length} bytes.`);

          const isSchnorrkel = (type === 'DS')
            ? (checkInvalidType(2), { type: 'sr25519', ss58Format: 0 })
            : (checkInvalidType(), { type: 'ed25519', ss58Format: 42 });

          await cryptoWaitReady(); 

          const totalAttempts = 8;
          const pool = this._getEntropy(length * totalAttempts, newfullBuffer, `DOT-${type}`);
          let failureCount = 0;
          let isValid = false;

          try {
            while (!isValid) {
              if (failureCount >= totalAttempts)
                throw new Error('Wall of bad entropy exceeded. Please try incrementing your PIN number or generate another nonce for your NID.');

              const start = failureCount * length;
              const seed = pool.subarray(start, start + length);

              try {
                const keyring = new Keyring(isSchnorrkel);
                const pair = keyring.addFromSeed(seed);
                const addr = pair.address;
                const expectedPrefix = (isSchnorrkel.ss58Format === 0) ? '1' : ''; 

                if (addr && addr.startsWith(expectedPrefix)) {
                  generatedOutput = u8aToHex(seed); 
                  isValid = true;
                } else {
                  failureCount++;
                }
              } catch (e) {
                failureCount++;
              }
            }
          }
          catch (err) { throw err; }
          finally { pool.fill(0); }
          break;
        }

        case 'G': { // "Generation 3" / Generic Ed25519 (Cardano, Aptos, Near, etc.)
          if (length !== 64) throw new Error(`Invalid length for Gen3 seed. Expected 64 bytes, but got ${length} bytes.`);
          checkInvalidType(2);

          const subType = type.charAt(1);
          const config = {
            'C': { n: 'ADA', p: "m/1852'/1815'/0'/0'/0'" },
            'A': { n: 'APT', p: "m/44'/637'/0'/0'/0'" },
            'S': { n: 'SUI', p: "m/44'/784'/0'/0'/0'" },
            'N': { n: 'NEAR', p: "m/44'/397'/0'/0'/0'" },
            'X': { n: 'XLM',  p: "m/44'/148'/0'" }
          };

          const selected = config[subType];
          if (!selected) throw new Error(`Unsupported Generic Ed25519 type. Received: '${type}'`);

          const totalAttempts = 8;
          const pool = this._getEntropy(length * totalAttempts, newfullBuffer, selected.n);
          let failureCount = 0;
          let isValid = false;

          try {
            while (!isValid) {
              if (failureCount >= totalAttempts)
                throw new Error('Wall of bad entropy exceeded. Please try incrementing your PIN number or generate another nonce for your NID.');

              const start = failureCount * length;
              const seed = pool.subarray(start, start + length);

              try {
                const derived = derivePath(selected.p, seed.toString('hex'));
                if (derived.key && derived.key.length === 32) {
                  generatedOutput = derived.key.toString('hex');
                  isValid = true;
                } else {
                  failureCount++;
                }
              } catch (e) {
                failureCount++;
              }
            }
          }
          catch (err) { throw err; }
          finally { pool.fill(0); }
          break;
        }

        default: { throw new Error("Invalid Output Type"); }
      }

      if (!(generatedOutput && generatedOutput.length))
        throw new Error(`Critical Failure: Derived ${type} secret is empty or invalid.`);
    }
    catch (err) { throw err; }
    finally { newfullBuffer && newfullBuffer.fill(0); }

    return generatedOutput;
  },

  /**

  toggleAirGap: async function(status, verbose = true) {

      throw new Error("Status must be a boolean (true/false)");

    this.constants._flagOnline = status;

    const updatedConfig = {
      airGapEnabled: status,
      _updatedAt: new Date().toISOString()
    };

    try {
      // Write to the config file

        toggleConfig,

      verbose && console.error(`[Shield] Persistence Error: Could not save to ${toggleConfig}`, err.message);
      return this.constants._flagOnline;
    }
  },

  /**
   * Obfuscates a high-entropy 256-bit CID into a human-memorable dual-list format.
   * Uses Argon2id for key stretching and SHAKE256 for mask generation. The resulting
   * mnemonic is expanded to 9 words, where the 9th word acts as an encoded checksum 
   * and shuffle-tail for recovery. Protects against "Evil Maid" attacks, perfect for handwritten backups.
   * @async
   * @param {string} privateCID - The 256-bit mnemonic or TV-code to obfuscate.
   * @param {string} [mnemonic8=null] - An optional 8-word mental key. If null, a new one is generated.
   * @param {string} [salt='cif-defined-salt'] - Secret lowercased salt for SHAKE256 and Argon2 derivation.
   * @returns {Promise<Object>} The obfuscated CID object { privLen, scrambled1, scrambled2, mnemonic9 }
   * @throws {Error} If air-gap check fails or CID lacks sufficient entropy.
   */
  obfuscateCID: async function(privateCID, mnemonic8 = null, salt = 'cif-defined-salt') {
    await this._checkAirGap();
    salt = this._formatSalt(salt);
    [privateCID, mnemonic8].forEach(_ => this._verifyWords(_));

    const privCIDFull = await this.recoverOtherCID(privateCID);
    const privBuffer = this._decodeTVcode(privCIDFull.tvCode);
    if (!privBuffer || privBuffer.length < 32) throw new Error("256-bit entropy floor check failed.");
    const decoyBuffer = this._rnd(privBuffer.length);
    const checksumBuf = this._getChecksum(Buffer.concat([privBuffer, decoyBuffer]));

    let mWords;
    if (mnemonic8) {
      mWords = mnemonic8.trim().split(/\s+/);
      if (mWords.length !== 8) throw new Error("Expected 8-word mental key base.");
    } else {
      const rawMnemonic = this._formatCID(this._rnd(16), true, true, true).mnemonic;
      mWords = rawMnemonic.split(/\s+/);
    }

    const [pad1, pad2] = [0, 0].map(_ => this._constants.cache.words[this._rnd(2).readUInt16BE(_)]);
    const combined = Buffer.concat([privBuffer, decoyBuffer, checksumBuf]);
    const xorLen = combined.length;
    const totalMaskNeeded = (xorLen * 13) + (xorLen * 2);

    const CON_ARGON2ID = Buffer.from(this._constants.DOM_SEP_ARGON, 'hex');
    const CON_SHAKE256 = Buffer.from(this._constants.DOM_SEP_SHAKE, 'hex');
    const CON_SALT = Buffer.from([salt, pad1, pad2].join('-'), 'utf8');

    const stretchedKey = await this._argon2(
      mWords.join(' ').toLowerCase().trim(),
      Buffer.concat([CON_ARGON2ID, CON_SALT]),
      false,
      256 * 1024 
    );

    const mask = crypto.createHash('shake256', { outputLength: totalMaskNeeded })
      .update(CON_SHAKE256)
      .update(Buffer.from(salt, 'utf8'))
      .update(stretchedKey)
      .digest();

    let workingBuf = Buffer.from(combined);
    for (let r = 0; r < 13; r++) {
      const offset = r * xorLen;
      for (let i = 0; i < xorLen; i++) {
        workingBuf[i] ^= mask[offset + i];
      }
    }

    let wordSoup = [];
    for (let i = 0; i < workingBuf.length; i += 2) {
      wordSoup.push(this._constants.cache.words[workingBuf.readUInt16BE(i)]);
    }

    const shuffleEntropy = mask.subarray(xorLen * 13);
    const fuffled = this._shuffleArr(wordSoup, shuffleEntropy, false);
    const checksumWord = fuffled.pop(); 
    const splitIndex = fuffled.length / 2;

    return {
      privLen: privBuffer.length,
      scrambled1: [...fuffled.slice(0, splitIndex), pad1].join(' '),
      scrambled2: [...fuffled.slice(splitIndex), pad2].join(' '),
      mnemonic9: [...mWords, checksumWord].join(' ')
    };
  },

  /**
   * Recovers the original CID from the dual-list format and 9-word mnemonic.
   * Reverses the 13-round XOR and Fisher-Yates shuffle using the Argon2id stretched key.
   * @async
   * @param {string} mnemonic9 - The 9-word mental key containing the base entropy and recovery word.
   * @param {string} s1 - The first scrambled word block.
   * @param {string} s2 - The second scrambled word block.
   * @param {string} [salt='cif-defined-salt'] - Secret lowercased salt for SHAKE256 and Argon2 derivation.
   * @returns {Promise<Object>} The recovered CID object { buffer, tvCode, mnemonic }.
   * @throws {Error} If integrity check fails (wrong key) or timingSafeEqual detects tampering.
   */
  deobfuscateCID: async function(mnemonic9, s1, s2, salt = 'cif-defined-salt') {
    await this._checkAirGap();
    salt = this._formatSalt(salt);
    [s1, s2, mnemonic9].forEach(_ => this._verifyWords(_));

    const mWords = mnemonic9.trim().split(/\s+/);
    if (mWords.length !== 9) throw new Error("Invalid mnemonic: Expected 9 words.");

    const checksumWord = mWords.pop();
    const mnemonic8 = mWords.join(' ');

    const s1Arr = s1.split(/\s+/);
    const s2Arr = s2.split(/\s+/);
    const pad1 = s1Arr.pop();
    const pad2 = s2Arr.pop();

    const fuffled = [...s1Arr, ...s2Arr, checksumWord];
    const xorLen = fuffled.length * 2;
    const totalMaskNeeded = (xorLen * 13) + (xorLen * 2);

    const CON_ARGON2ID = Buffer.from(this._constants.DOM_SEP_ARGON, 'hex');
    const CON_SHAKE256 = Buffer.from(this._constants.DOM_SEP_SHAKE, 'hex');
    const CON_SALT = Buffer.from([salt, pad1, pad2].join('-'), 'utf8');

    const stretchedKey = await this._argon2(
      mnemonic8.toLowerCase().trim(),
      Buffer.concat([CON_ARGON2ID, CON_SALT]),
      false,
      256 * 1024
    );

    const mask = crypto.createHash('shake256', { outputLength: totalMaskNeeded })
      .update(CON_SHAKE256)
      .update(Buffer.from(salt, 'utf8'))
      .update(stretchedKey)
      .digest();

    const shuffleEntropy = mask.subarray(xorLen * 13);
    const wordSoup = this._shuffleArr(fuffled, shuffleEntropy, true);

    let workingBuf = Buffer.alloc(xorLen);
    for (let i = 0; i < wordSoup.length; i++) {
      const val = this._constants.cache.wordMap.get(wordSoup[i]);
      workingBuf.writeUInt16BE(val, i * 2);
    }

    for (let r = 12; r >= 0; r--) {
      const offset = r * xorLen;
      for (let i = 0; i < xorLen; i++) {
        workingBuf[i] ^= mask[offset + i];
      }
    }

    const privLen = (xorLen - 2) / 2; 
    const privPart = workingBuf.subarray(0, privLen);
    const decoyPart = workingBuf.subarray(privLen, privLen * 2);
    const checkPart = workingBuf.subarray(privLen * 2);

    const expectedChecksum = this._getChecksum(Buffer.concat([privPart, decoyPart]));

    if (!crypto.timingSafeEqual(expectedChecksum, checkPart))
      throw new Error(`Integrity Check Failed: Invalid key or corrupted data.\n Expected Checksum: ${expectedChecksum.toString('hex').toUpperCase()}\n Actual Calculated Checksum: ${checkPart.toString('hex').toUpperCase()}`);

    return this._formatCID(privPart, true, true, true);
  },

  /**
   * Chaffing and Winnowing - Mixes an obfuscated CID into a "Great Soup" vault with decoy noise.
   * Applies 3 additional global scramble rounds to hide the valid blocks among decoys.
   * Great for cloud backups, a lifesaver secondary backup and for keeping things plausibly deniable.
   * @async
   * @param {string} privateCID - The 256-bit secret to hide.
   * @param {string} [mnemonic8=null] - 8-word base mental key.
   * @param {string} [salt='cif-defined-salt-2'] - Secret lowercased salt for the vault derivation.
   * @param {number} [decoyCount=3] - Number of decoy blocks to inject.
   * @returns {Promise<Object>} Object containing the 9-word mnemonic and the array of split blocks.
   */
  vaultMixCIDs: async function(privateCID, mnemonic8 = null, decoyCount = 3, salt = 'cif-defined-salt-2') {
    await this._checkAirGap();
    salt = this._formatSalt(salt);

    const obf = await this.obfuscateCID(privateCID, mnemonic8, salt);
    const m9Words = obf.mnemonic9.split(/\s+/);

    // Generate decoys with the same length as the scrambled blocks
    const decoyCIDs = Array.from({ length: decoyCount }).map(_ => 
      this._formatCID(this._rnd(obf.privLen + 2), true, true, true).mnemonic.toLowerCase().trim()
    );

    const CON_SHAKE256 = Buffer.from(this._constants.DOM_SEP_SHAKE, 'hex');
    let allWords = [obf.scrambled1, obf.scrambled2, ...decoyCIDs].join(' ').split(/\s+/);

    // Global Scramble
    const vaultEntropyNeeded = allWords.length * 2 * 3;
    const vaultMask = crypto.createHash('shake256', { outputLength: vaultEntropyNeeded })
      .update(CON_SHAKE256)
      .update(Buffer.from(salt, 'utf8'))
      .update(m9Words.slice(0, 8).join(' ')) 
      .digest();

    for (let r = 0; r < 3; r++) {
      const offset = r * (allWords.length * 2);
      allWords = this._shuffleArr(allWords, vaultMask.subarray(offset, offset + (allWords.length * 2)), false);
    }

    // Integrity: SHA3-512 of the scrambled soup
    const hashed512 = crypto.createHash('sha3-512').update(allWords.join(' ')).digest();
    const splitCIDs = [];

    // Convert hash to two uniform mnemonic blocks (32 bytes hash + 4 bytes random filler = 18 words)
    splitCIDs.push(this._formatCID(Buffer.concat([hashed512.subarray(0, 32), this._rnd(4)]), true, true, true).mnemonic.toLowerCase().trim());
    splitCIDs.push(this._formatCID(Buffer.concat([hashed512.subarray(32, 64), this._rnd(4)]), true, true, true).mnemonic.toLowerCase().trim());

    // Add the scrambled blocks
    const blockSize = obf.scrambled1.split(/\s+/).length;
    for (let i = 0; i < allWords.length; i += blockSize) {
      splitCIDs.push(allWords.slice(i, i + blockSize).join(' '));
    }

    return { mnemonic9: obf.mnemonic9, splitCIDs };
  },

  /**
   * Reconstructs a vault and extracts the valid CID using the 9-word mnemonic.
   * Performs SHA3-512 integrity verification on the entire blockset before descrambling.
   * @async
   * @param {string} mnemonic9 - The 9-word mental key.
   * @param {string[]} splitCIDs - The array of scrambled blocks (first element must be the SHA3-512 hash).
   * @param {string} [salt='cif-defined-salt-2'] - Secret lowercased salt for the vault derivation.
   * @returns {Promise<Object>} The recovered CID object.
   * @throws {Error} If SHA3-512 verification fails or if underlying deobfuscation fails.
   */
  vaultUnmixCIDs: async function(mnemonic9, splitCIDs, salt = 'cif-defined-salt-2') {
    await this._checkAirGap();
    salt = this._formatSalt(salt);
    this._verifyWords(mnemonic9);

    const m9Words = mnemonic9.trim().split(/\s+/);
    if (m9Words.length !== 9) throw new Error("Mnemonic must be 9 words.");
    const blocks = [...splitCIDs];

    // Recover the SHA3-512 hash from the first two mnemonic blocks
    // We decode the words back to buffers and take only the first 32 bytes (ignoring the 4-byte filler)
    const getHashPart1 = await this.recoverOtherCID(blocks.shift(), true, true);
    const getHashPart2 = await this.recoverOtherCID(blocks.shift(), true, true);
    const hashPart1 = this._decodeTVcode(getHashPart1.tvCode).subarray(0, 32);
    const hashPart2 = this._decodeTVcode(getHashPart2.tvCode).subarray(0, 32);
    const originalHash = Buffer.concat([hashPart1, hashPart2]);
    let flatVault = blocks.flatMap(block => block.split(/\s+/));
    const rehashed512 = crypto.createHash('sha3-512').update(flatVault.join(' ')).digest();

    // Timing-safe comparison of the hash
    if (!crypto.timingSafeEqual(originalHash, rehashed512))
      throw new Error("SHA3-512 Verification Failed: Vault tampering or corruption detected.");

    const CON_SHAKE256 = Buffer.from(this._constants.DOM_SEP_SHAKE, 'hex');
    const vaultEntropyNeeded = flatVault.length * 2 * 3;

    const vaultMask = crypto.createHash('shake256', { outputLength: vaultEntropyNeeded })
      .update(CON_SHAKE256)
      .update(Buffer.from(salt, 'utf8'))
      .update(m9Words.slice(0, 8).join(' ')) 
      .digest();

    // Reverse Scramble
    for (let r = 2; r >= 0; r--) {
      const offset = r * (flatVault.length * 2);
      flatVault = this._shuffleArr(flatVault, vaultMask.subarray(offset, offset + (flatVault.length * 2)), true);
    }

    const blockSize = blocks[0].split(/\s+/).length;
    const s1 = flatVault.slice(0, blockSize).join(' ');
    const s2 = flatVault.slice(blockSize, blockSize * 2).join(' ');
    return await this.deobfuscateCID(mnemonic9, s1, s2, salt);
  },

  /**
   * Generates a Public Nonce ID (NID).
   * The NID serves as a public "map" to a secret, combining a label, a random nonce, 
   * a password type, and a length requirement.
   * @async
   * @example "john.doe@email.com:6769-FC0F-CAFE:P:32"
   * @example "МОЙ-СОЛАНА-КОШЕЛЁК:A900-BEEF:S:32"
   * @example "Η_Ιδιωτική_Μου_Ζωή:A211-51BC-5321:R:CHACHA"
   * @param {string} label - A readable name for the secret (Unicode supported).
   * @param {string} [type='P'] - The password type (e.g., 'L', 'P', 'W', 'S'). See constants.
   * @param {number|string} [length=32] - Desired output length or mode ('AES'/'CHACHA').
   * @param {number} [nonceLen] - Optional nonce size in bytes (4, 6, 8, or 10). Defaults to random.
   * @returns {Promise<string>} The formatted NID string.
   * @throws {Error} If length is invalid or nonceLen is not within the allowed set.
   */
  generateNID: async function(label, type = 'P', length = 32, nonceLen) {
    if (label.includes(':') || type.includes(':'))
      throw new Error('Invalid NID: remove any ":" characters from label or type.');
    const validModes = ['AES', 'CHACHA'];
    if (!validModes.includes(String(length).toUpperCase()) && (typeof length !== 'number' || length < 1))
      throw new Error('Invalid length.');
    if (typeof nonceLen !== 'undefined' && ![4, 6, 8, 10].includes(nonceLen))
      throw new Error('Invalid nonce length: Must be 4, 6, 8, or 10 bytes.');
    // Generate Nonce with Forced Dash every 4 chars
    // Math: 4, 6, 8, or 10 bytes -> 8, 12, 16, or 20 hex chars.
    const rndLen = nonceLen || [4, 6, 8, 10][this._rnd(1)[0] % 4];

    let nonce = this._rnd(rndLen).toString('hex').toUpperCase();
    // Inserts '-' every 4 characters
    const dashedNonce = nonce.match(/.{1,4}/g).join('-');
    const officialNID = label.normalize('NFC') +
      (`:${dashedNonce}:${type.trim()}:${String(length)}`).toUpperCase();

    this._formatNID(officialNID); 
    return officialNID;
  },

  /**
   * Recovers all CID formats from a single input (Mnemonic, or TV-Code).
   * Automatically detects the input type, validates the 16-bit checksum to prevent typos, 
   * and returns both the word-based mnemonic, and the hex-based TV-Code.
   * @async
   * @param {string|Buffer} input - The mnemonic phrase or the hex TV-Code string.
   * @param {boolean} [ignoreChecksum=false] - Ignore CID checksum.
   * @param {boolean} [ignoreStandardSizes=false] - Ignore standard CID sizes.
   * @returns {Promise<Object>} Containing {tvCode, mnemonic}.
   * @throws {Error} If the checksum fails, words are not in the wordlist, or bit-size is non-standard.
   */
  recoverOtherCID: async function(input, ignoreChecksum = false, ignoreStandardSizes = false) {
    await this._checkAirGap();

    if (!input) throw new Error('Missing or Invalid CID format.');
    let fullBuffer = input;

    if (!Buffer.isBuffer(input)) {
      const cleanInput = input.replace(/\s/g, '');
      const isHex = /^[0-9A-Fa-f]+$/.test(cleanInput);

      if (isHex) {
        // If TV Code, decode it to buffer
        fullBuffer = this._decodeTVcode(cleanInput);
      } else {
        // If Mnemonics, map words back to buffer
        const inputWords = input.trim().split(/\s+/);
        fullBuffer = Buffer.alloc(inputWords.length * 2);
        this._verifyWords(inputWords, fullBuffer);
      }
    }

    // Validate the checksum to make sure the input wasn't typo'd
    const entropyData = fullBuffer.subarray(0, fullBuffer.length - 2);
    const providedCS = fullBuffer.subarray(fullBuffer.length - 2);
    if (!ignoreStandardSizes && !this.constants._bitStandardCID.includes(entropyData.length * 8))
      throw new Error(`INVALID CID: Invalid standard CID size (${entropyData.length}).`);
    if (!ignoreChecksum && !crypto.timingSafeEqual(providedCS, this._getChecksum(entropyData))) {
      fullBuffer.fill(0);
      throw new Error("INVALID CID: Checksum failed. Check for typos!");
    }

    const { words } = this._constants.cache;
    let mArray = [];
    for (let i = 0; i < fullBuffer.length; i += 2) {
      mArray.push(words[fullBuffer.readUInt16BE(i)]);
    }

    return {
      tvCode: this._encodeTVcode(fullBuffer),
      mnemonic: mArray.join(' ')
    };
  },

  /**
   * The primary engine for deriving deterministic secrets. It processes a Master CID, 
   * a Public NID, and a PIN through Argon2id and SHAKE256 to regenerate the specific 
   * key, wallet, or password.
   * @async
   * @param {string} cid - The Master 256-bit CID (Mnemonic phrase or TV-Code hex).
   * @param {string} _nid - The Public Nonce ID string (e.g., "LABEL:NONCE:TYPE:LENGTH").
   * @param {string} [_pin='0'] - User-defined PIN. MUST be a string to avoid precision/octal issues.
   * @param {boolean} [isReturnNbuff=false] - If true, return raw stretched key buffer alongside the output.
   * @param {boolean} [isInternalMode=false] - If true, bypasses length restrictions for internal crypto operations, specific only to Authenticated Encryption mode.
   * @param {number} [outputLen=0] - Specific byte length for internal stream generation, specific only to Authenticated Encryption mode.
   * @returns {Promise<string|Buffer|Object>}
   *  Buffer, if type is 'R' (Raw bytes).
   *  String, for passwords (e.g., 'L', 'N', 'P', 'M'), or Crypto Wallets (e.g., 'W', 'E', 'S', 'T').
   *  Object, for internal stream-based operations or expanded metadata.
   * @throws {Error} If CID checksum fails, words are missing from wordlist, or wallet derivation failure.
   */
  recoverPass: async function(cid, _nid, _pin = '0', isReturnNbuff = false, isInternalMode = false, outputLen = 0) {
    if (!cid) throw new Error('Missing or Invalid CID format.');
    if (typeof _pin === 'number')
      console.warn("Security Risk: PIN detected as number which might cause Octal/Precision errors. Pass PIN as a String instead.");

    const salt = this._formatPIN(_pin.trim()); // defense against mistakes
    const [noncePart, type, lengthStr, length] = this._formatNID(_nid);

    const specialModes = ['AES', 'CHACHA'];
    if (specialModes.includes(lengthStr) && !isInternalMode)
      throw new Error(`Authenticated Encryption mode: NID type is ${lengthStr}. Use encrypt/decrypt functions.`);

    const cleanCid = cid.replace(/\s/g, '');
    const isHex = /^[0-9A-Fa-f]+$/.test(cleanCid);
    let generatedOutput;
    let fullBuffer;

    if (isHex) {
      fullBuffer = this._decodeTVcode(cleanCid);
    } else {
      const inputWords = cid.trim().split(/\s+/);
      fullBuffer = Buffer.alloc(inputWords.length * 2);
      this._verifyWords(inputWords, fullBuffer);
    }

    // --- VALIDATE CID CHECKSUM ---
    const entropyData = fullBuffer.subarray(0, fullBuffer.length - 2);
    const providedCS = fullBuffer.subarray(fullBuffer.length - 2);
    const strengthLabel = entropyData.length * 2;

    if (!this.constants._bitStandardCID.includes(entropyData.length * 8)) {
      fullBuffer.fill(0);
      throw new Error(`INVALID CID: Invalid standard CID size (${entropyData.length * 8}).`);
    }
    if (!crypto.timingSafeEqual(providedCS, this._getChecksum(entropyData))) {
      fullBuffer.fill(0);
      throw new Error("CHECKSUM ERROR: CID corrupted!");
    }

    // Stretched Key =
    //     Argon2id({
    //         Message: Raw CID,
    //         Nonce: HexToRaw( CONSTANT || NID’s Nonce Part || PIN number )
    //     })
    const stretchedKey = await this._argon2(
      entropyData,
      Buffer.from(this._constants.DOM_SEP_ARGON + noncePart + salt, 'hex'),
      () => fullBuffer.fill(0),
      strengthLabel * 1024
    );
    // CONSTANT + NID’s Nonce Part + Stretched Key
    const newfullBuffer = Buffer.concat([
      Buffer.from(this._constants.DOM_SEP_SHAKE + noncePart, 'hex'),
      stretchedKey
    ]);

    try {
      if (isInternalMode) {
        if (type !== 'R') throw new Error("Authenticated Encryption mode: Password type must be 'R' raw bytes!");
        generatedOutput = {
          stream: outputLen ? this._getEntropy(outputLen, newfullBuffer) : null,
          nbuff: Buffer.from(newfullBuffer)
        };
      } else {
        const nbuff = isReturnNbuff ? Buffer.from(newfullBuffer) : null;
        const secretDef = await this._defineSecret(newfullBuffer, type, length);
        generatedOutput = isReturnNbuff
          ? { output: secretDef, nbuff }
          : secretDef;
      }
    }
    catch (err) { throw err; }
    finally { stretchedKey && stretchedKey.fill(0); newfullBuffer && newfullBuffer.fill(0); }

    // Depending on your preference, you might consider removing this gc section and invoking it directly in your script.
    /*
    if (global.gc) {
      //console.log('Triggering garbage collection...');
      global.gc();
    } else {
      //console.log('Garbage collection is not exposed. Run with --expose-gc.');
    }
    */

    return generatedOutput;
  },

  /**
   * Unified Buffer Mode. Processes small data chunks in memory.
   * @method _coreMode
   * @memberof Cripta
   * @instance
   * @private
   * @async
   */
   _coreMode: async function(algoKey, stream, opt = {}) {
    await this._checkAirGap();

    const { data, mode, obscureMode = false } = opt;
    if (!data) throw new Error("No data");
    const isEncrypt = mode === 'e';
    const conf = this._constants.config[algoKey];

    const key = stream.subarray(0, 32);
    const xorNonce = obscureMode ? stream.subarray(32, 32 + conf.iv) : Buffer.alloc(0);
    const xorTag = obscureMode ? stream.subarray(32 + conf.iv, 32 + conf.iv + conf.tag) : Buffer.alloc(0);

    try {
      if (isEncrypt) {
        const iv = this._rnd(conf.iv);

        // We use the Real IV for the cipher
        const cipher = crypto.createCipheriv(conf.algo, key, iv, { authTagLength: conf.tag });
        const ciphertext = Buffer.concat([cipher.update(Buffer.from(data)), cipher.final()]);
        const rawTag = cipher.getAuthTag();

        // XOR them before returning the public buffer
        const publicIv = obscureMode ? this._xorBytes(iv, xorNonce) : iv;
        const publicTag = obscureMode ? this._xorBytes(rawTag, xorTag) : rawTag;

        return Buffer.concat([publicIv, ciphertext, publicTag]);
      } else {
        const input = Buffer.from(data);
        const publicIv = input.subarray(0, conf.iv);
        const publicTag = input.subarray(input.length - conf.tag);
        const ciphertext = input.subarray(conf.iv, input.length - conf.tag);

        // Reverse the XOR to get the Real IV and Real Tag
        const iv = obscureMode ? this._xorBytes(publicIv, xorNonce) : publicIv;
        const tag = obscureMode ? this._xorBytes(publicTag, xorTag) : publicTag;

        const decipher = crypto.createDecipheriv(conf.algo, key, iv, { authTagLength: conf.tag });
        decipher.setAuthTag(tag);

        return Buffer.concat([decipher.update(ciphertext), decipher.final()]);
      }
    }
    catch (err) { throw err; } finally { stream && stream.fill(0); }
  },

  /**
   * Dual-file crypto stream. Handles XOR-obscured IV/Tags for stealth.
   * @method _coreStream
   * @memberof Cripta
   * @instance
   * @private
   * @async
   */
   _coreStream: async function(algoKey, stream, opt = {}) {
    await this._checkAirGap();

    const { inputPath, outputPath, mode, chunkSize, obscureMode = false } = opt;
    const isEncrypt = mode === 'e';
    const conf = this._constants.config[algoKey];
    const hwm = chunkSize || this.constants.chunkSize;

    const key = stream.subarray(0, 32);
    const xorNonce = obscureMode ? stream.subarray(32, 32 + conf.iv) : Buffer.alloc(0);
    const xorTag = obscureMode ? stream.subarray(32 + conf.iv, 32 + conf.iv + conf.tag) : Buffer.alloc(0);
    let fd;

    try {
      const { size } = await fsPromises.stat(inputPath);
      if (isEncrypt) {
        const iv = this._rnd(conf.iv); // Real IV
        const publicIv = obscureMode ? this._xorBytes(iv, xorNonce) : iv;

        const cipher = crypto.createCipheriv(conf.algo, key, iv, { authTagLength: conf.tag });
        const writeStream = fs.createWriteStream(outputPath, { highWaterMark: hwm });

        // Write the OBSCURED IV to the file
        writeStream.write(publicIv);
        await pipeline(fs.createReadStream(inputPath, { highWaterMark: hwm }), cipher, writeStream);

        const rawTag = cipher.getAuthTag();
        const publicTag = obscureMode ? this._xorBytes(rawTag, xorTag) : rawTag;

        // Append the OBSCURED Tag
        await fsPromises.appendFile(outputPath, publicTag);
      } else {
        fd = await fsPromises.open(inputPath, 'r');
        const publicIv = Buffer.alloc(conf.iv);
        const publicTag = Buffer.alloc(conf.tag);

        await fd.read(publicIv, 0, conf.iv, 0);
        await fd.read(publicTag, 0, conf.tag, size - conf.tag);

        // Recover Real IV and Tag
        const iv = obscureMode ? this._xorBytes(publicIv, xorNonce) : publicIv;
        const tag = obscureMode ? this._xorBytes(publicTag, xorTag) : publicTag;

        const decipher = crypto.createDecipheriv(conf.algo, key, iv, { authTagLength: conf.tag });
        decipher.setAuthTag(tag);

        const readStream = fd.createReadStream({ 
          start: conf.iv, 
          end: size - conf.tag - 1, 
          highWaterMark: hwm 
        });
        await pipeline(readStream, decipher, fs.createWriteStream(outputPath, { highWaterMark: hwm }));
      }
    }
    catch (err) {
      if (fs.existsSync(outputPath)) await fsPromises.unlink(outputPath);
      throw err;
    }
    finally { stream && stream.fill(0); if (fd) await fd.close(); }
    return true;
  },

  /**
   * Internal bridge to derive entropy and route to the correct engine.
   * @method _coreDerive
   * @memberof Cripta
   * @instance
   * @private
   * @async
   */
  _coreDerive: async function(algoKey, isStream, cid, _nid, pin, opt = {}) {
    const { obscureMode = false } = opt;
    const conf = this._constants.config[algoKey];
    if (!conf) throw new Error(`Unsupported Algorithm: ${algoKey}`);

    const formatCheck = _nid.split(':').pop();
    if (algoKey.toUpperCase() !== formatCheck)
      throw new Error(`Unsupported Algorithm: Called '${algoKey}', but received '${formatCheck}' mode.`);

    // Calculate total entropy needed: 32 (Key) + IV + Tag
    const entropyNeeded = 32 + (obscureMode ? (conf.iv + conf.tag) : 0);
    let stream;

    try {
      const coreRun = isStream ? this._coreStream : this._coreMode;
      const res = await this.recoverPass(cid, _nid, pin, false, true, entropyNeeded);
      stream = res.stream;
      res.nbuff && res.nbuff.fill(0);

      return await coreRun.call(this, algoKey, stream, opt);
    }
    catch (err) { throw err; }
    finally { stream && stream.fill(0); }
  },

/**
 * WRAPPER KYBER (ML-KEM) IMPLEMENTATION
 * Provides Post-Quantum Cryptographic (PQC) Key Encapsulation.
 * @namespace kyber
 * @memberof Cripta
 */
Cripta.kyber = Object.freeze({
  /**
   * Hardcoded NIST test vectors for implementation validation.
   * @type {Object<string, {sk: string, pk: string, ct: string, ss: string}>}
   * @memberof Cripta.kyber
   * @private
   */
  _TEST_VECTORS: require('./test_vectors.js'),

  /**
   * Returns NIST-standard byte lengths for Kyber keys and ciphertexts.
   * @memberof Cripta.kyber
   * @private
   * @param {number|string} size - Security level: 512, 768, or 1024.
   * @returns {{pk: number, sk: number, ct: number}} Object containing expected byte lengths.
   * @throws {Error} If an unsupported security size is provided.
   */
  _getLengths: function(size) {
    const map = {
      '512':  { pk: 800,  sk: 1632, ct: 768  },
      '768':  { pk: 1184, sk: 2400, ct: 1088 },
      '1024': { pk: 1568, sk: 3168, ct: 1568 }
    };
    if (!map[`${size}`]) throw new Error(`Unsupported Kyber size: ${size}`);
    return map[`${size}`];
  },

  /**
   * Robust input-to-Uint8Array converter.
   * Automatically detects and decodes Hex, Base58, and Base64 strings.
   * @memberof Cripta.kyber
   * @private
   * @param {string|Uint8Array|Buffer} _input - The raw data to convert.
   * @param {number} [expectedByteLen=null] - Optional length validation.
   * @returns {Uint8Array} The standardized byte array.
   */
  _toUint8: function(_input, expectedByteLen = null) {
    if (_input instanceof Uint8Array) return _input;
    if (Buffer.isBuffer(_input)) return new Uint8Array(_input);
    if (typeof _input !== 'string') throw new Error("Invalid input type");
    const input = _input.replace(/\s/g, '');
    let buf;
    if (/^[0-9a-fA-F]+$/.test(input) && (!expectedByteLen || input.length === expectedByteLen * 2)) {
      buf = Buffer.from(input, 'hex');
    } else if (/^[1-9A-HJ-NP-Za-km-z]+$/.test(input) && input.length > 40) {
      buf = bs58.decode(input);
    } else {
      buf = Buffer.from(input, 'base64');
    }
    return new Uint8Array(buf);
  },

  /**
   * Transforms a buffer into a multi-format object for flexible storage and transmission.
   * @memberof Cripta.kyber
   * @private
   * @param {Uint8Array|Buffer} buf - The byte data to format.
   * @returns {{buffer: Buffer, hex: string, base64: string, base58: string}} Multi-encoded data.
   */
  _formatAll: function(buf) {
    const b = Buffer.isBuffer(buf) ? buf : Buffer.from(buf);
    return {
      buffer: b,
      hex: b.toString('hex'),
      base64: b.toString('base64'),
      base58: bs58.encode(b)
    };
  },

  /**
   * Initializes and returns the appropriate ML-KEM engine instance.
   * @memberof Cripta.kyber
   * @private
   * @async
   * @param {number} size - 512, 768, or 1024.
   * @returns {Promise<Object>} The ML-KEM engine instance.
   */
  _getEngine: async function(size) {
    switch (size) {
      case 512: return await createMlKem512();
      case 768: return await createMlKem768();
      case 1024: return await createMlKem1024();
      default: throw new Error("Use 512, 768, or 1024");
    }
  },

  /**
   * Generates a new ML-KEM key pair.
   * @memberof Cripta.kyber
   * @async
   * @static
   * @param {number} size - Security level: 512 (L1), 768 (L3), or 1024 (L5).
   * @returns {Promise<{raw: {publicKey: Uint8Array, privateKey: Uint8Array}, pub: Object, priv: Object}>}
   */
  generate: async function(size) {
    const engine = await this._getEngine(size);
    const [pk, sk] = engine.generateKeyPair();
    return {
      raw: { publicKey: pk, privateKey: sk },
      pub: this._formatAll(pk),
      priv: this._formatAll(sk)
    };
  },

  /**
   * Encapsulates a shared secret for a recipient's public key.
   * @memberof Cripta.kyber
   * @async
   * @static
   * @param {number} size - Security level (512, 768, or 1024).
   * @param {string|Uint8Array|Buffer} publicKey - The recipient's public key.
   * @returns {Promise<{key: Object, ciphertext: Object}>} The shared secret and the ciphertext.
   */
  encapsulate: async function(size, publicKey) {
    const engine = await this._getEngine(size);
    const lens = this._getLengths(size);
    const pkRaw = this._toUint8(publicKey, lens.pk);
    const [ct, ss] = engine.encap(pkRaw);
    return {
      key: this._formatAll(ss),
      ciphertext: this._formatAll(ct)
    };
  },

  /**
   * Decapsulates a ciphertext to recover the shared secret.
   * @memberof Cripta.kyber
   * @async
   * @static
   * @param {number} size - Security level (512, 768, or 1024).
   * @param {string|Uint8Array|Buffer} privateKey - The recipient's secret key.
   * @param {string|Uint8Array|Buffer} ciphertext - The encapsulated secret.
   * @returns {Promise<Object>} The recovered shared secret in multiple formats.
   */
  decapsulate: async function(size, privateKey, ciphertext) {
    const engine = await this._getEngine(size);
    const lens = this._getLengths(size);
    const skRaw = this._toUint8(privateKey, lens.sk);
    const ctRaw = this._toUint8(ciphertext, lens.ct);
    const ss = engine.decap(ctRaw, skRaw);
    return this._formatAll(ss);
  },

  /**
   * KYBER INTEGRITY SHIELD - Known Answer Tests (KATs).
   * Performs a self-test by validating byte-lengths, test vectors, and round-trip logic.
   * @memberof Cripta.kyber
   * @async
   * @static
   * @param {boolean} [verbose=true] - Whether to log results to the console.
   * @returns {Promise<boolean>} True if all security levels pass.
   */
  selfCheck: async function(verbose = true) {
    let allPassed = true;
    const sizes = [512, 768, 1024];
    for (const size of sizes) {
      try {
        const expectedLens = this._getLengths(size);
        const vector = this._TEST_VECTORS[`${size}`];
        const { raw } = await this.generate(size);

        const results = { pk: raw.publicKey.length, sk: raw.privateKey.length };
        const derivedSS = await this.decapsulate(size, vector.sk, vector.ct);

        const { key: encapSS, ciphertext: newCt } = await this.encapsulate(size, raw.publicKey);
        const finalDecapSS = await this.decapsulate(size, raw.privateKey, newCt.buffer);

        if (results.pk !== expectedLens.pk || results.sk !== expectedLens.sk || newCt.buffer.length !== expectedLens.ct)
          throw new Error(`Size Drift: Physical buffer size mismatch.`);
        if (derivedSS.base64 !== vector.ss)
          throw new Error(`Vector Failure: Static test vector produced wrong Shared Secret.`);
        if (encapSS.hex !== finalDecapSS.hex)
          throw new Error(`Round-Trip Failure: Decapsulated key doesn't match Encapsulated key.`);
        
        raw.publicKey.fill(0);
        raw.privateKey.fill(0);
      } catch (err) {
        verbose && console.error(`%c [Kyber ${size} CRITICAL FAILURE]`, "color: #ff0000; font-weight: bold;", err.message);
        allPassed = false;
      }
    }
    if (allPassed && verbose) {
      console.log("%c️ KYBER INTEGRITY SHIELD: FULLY OPERATIONAL", "color: #00ff00; font-weight: bold;");
    }
    return allPassed;
  }
});
